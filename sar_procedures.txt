# ---------------------------------------------------------------------------
# Code-level section refinement
# For specific high-value codes, inject only the most relevant sections
# rather than the full typology group set
# ---------------------------------------------------------------------------

FINCEN_CODE_SECTION_REFINEMENT = {
    "821": [1, 2, 5, 16],        # source of funds — customer info + avoidance + inconsistent business
    "824": [3, 5, 13, 16],       # funnel account — funds transfers + inconsistent business + shell/pass-through
    "805": [3, 7, 8],            # EFT/wire — funds transfers + correspondent + cross-border
    "812": [5, 15, 16],          # out of pattern — inconsistent business + employees + other unusual
    "807": [3, 5, 13],           # multiple accounts — funds transfers + inconsistent + shell
    "114": [2],                  # below CTR threshold — avoidance only
    "113": [2],                  # below BSA threshold — avoidance only
    "925": [5, 16],              # no apparent purpose — inconsistent business + other unusual
    "911": [13, 16],             # two or more individuals — shell/straw-man + other unusual
}


# ---------------------------------------------------------------------------
# Main entry point
# ---------------------------------------------------------------------------

def get_procedures_for_tags(
    active_fincen_tags: List[Dict[str, Any]]
) -> str:
    """
    Returns formatted red flag indicator text for injection into
    LLM prompt, based on active FinCEN tags for a narrative.

    active_fincen_tags: list of dicts with keys:
        fincen_tag (str), typology (str), description (str)
        — same structure as built in agent.py induct() loop

    Returns:
        Formatted string of relevant KeyBank red flag indicators,
        deduplicated across tags. Empty string if no procedures found.
    """
    if not active_fincen_tags:
        return ""

    # Collect relevant section numbers, deduplicated
    sections_to_inject = set()

    for tag_entry in active_fincen_tags:
        code = str(tag_entry.get("fincen_tag", ""))

        # Code-level refinement takes priority over group-level
        if code in FINCEN_CODE_SECTION_REFINEMENT:
            for s in FINCEN_CODE_SECTION_REFINEMENT[code]:
                sections_to_inject.add(s)
        else:
            # Fall back to typology group sections
            code_info = FINCEN_CODE_MAP.get(code, {})
            typology_group = code_info.get("typology_group", "")
            for s in SAR_PROCEDURES_INDEX.get(typology_group, []):
                sections_to_inject.add(s)

    if not sections_to_inject:
        return ""

    # Build formatted output
    lines = ["Relevant KeyBank red flag indicators:"]

    for section_num in sorted(sections_to_inject):
        section = _RED_FLAG_SECTIONS.get(section_num)
        if not section:
            continue

        title = section.get("title", f"Section {section_num}")
        indicators = section.get("indicators", [])

        if not indicators:
            continue

        lines.append(f"\n  {title}:")
        for item in indicators:
            lines.append(f"    - {item['text']}")

    return "\n".join(lines)


# ---------------------------------------------------------------------------
# Code description lookup — used by agent.py for logging/display
# ---------------------------------------------------------------------------

def get_code_description(fincen_code: str) -> str:
    """Returns human readable description for a FinCEN code."""
    entry = FINCEN_CODE_MAP.get(str(fincen_code), {})
    return entry.get("description", f"Unknown code: {fincen_code}")


def get_code_typology_group(fincen_code: str) -> str:
    """Returns typology group string for a FinCEN code."""
    entry = FINCEN_CODE_MAP.get(str(fincen_code), {})
    return entry.get("typology_group", "")

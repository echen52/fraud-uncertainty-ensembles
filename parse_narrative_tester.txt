from tca.parse_narrative import parse_narrative
from tca.tools.sar_narrative_tools import get_sar_narratives

# Pull a small sample
df = get_sar_narratives(
    start_date="2026-01-01",
    end_date="2026-03-31"
)

# Test on first 5 narratives
for _, row in df.head(5).iterrows():
    narrative = row.get("NARRATIVE", "")
    if not narrative or len(str(narrative)) < 100:
        continue
    
    result = parse_narrative(narrative)
    
    print(f"Case ID: {row['ITEM_ID']}")
    print(f"  Suspicious Total: {result['suspicious_total']}")
    print(f"  Review Period: {result['review_period']}")
    print(f"  SAR Factors: {result['sar_factors']}")
    print(f"  Keyword Matches: {list(result['keyword_matches'].keys())}")
    print("-----")

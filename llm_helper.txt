def _ms(t0: float) -> int:
    return int((time.time() - t0) * 1000)


# --- Shared ADK Gemini call wrapper ---

async def _call_gemini(
    instruction: str,
    user_prompt: str,
    model: str = "gemini-2.5-flash",
) -> str:
    """
    Shared ADK Gemini call wrapper.
    Takes a system instruction and user prompt, returns response text.
    """
    session_id = f"session_{uuid.uuid4().hex[:8]}"

    agent = Agent(
        name="tca_llm_agent",
        model=model,
        instruction=instruction,
    )

    session_service = InMemorySessionService()
    await session_service.create_session(
        app_name="tca",
        user_id="tca_user",
        session_id=session_id,
    )

    runner = Runner(
        agent=agent,
        session_service=session_service,
        app_name="tca",
    )

    response_text = ""
    async for event in runner.run_async(
        user_id="tca_user",
        session_id=session_id,
        new_message=Content(parts=[Part(text=user_prompt)])
    ):
        if hasattr(event, 'content') and event.content:
            for part in event.content.parts:
                if hasattr(part, 'text') and part.text:
                    response_text += part.text

    return response_text


# --- Reading Plan ---

READING_PLAN_PROMPT = """
You are an AML expert tasked with defining a structured Reading Plan for analyzing 
SAR narratives. A Reading Plan is a formalized set of instructions that defines how 
to read and extract information from SAR narratives consistently across cases.

Given a sample of historical SAR narratives with their FinCEN typology tags, 
generate a Reading Plan that defines:
1. How to identify and parse each narrative section (Introduction, Details of 
   Investigation, Suspicious Activity List, Conclusion, Contents of File)
2. What key information to extract from each section
3. What behavioral patterns to look for that are indicative of each FinCEN typology
4. What contextual signals (counterparty relationships, channel types, geographic 
   risk, transaction velocity) are most relevant for typology classification

Return a structured JSON Reading Plan only, no other text.
"""


async def generate_reading_plan(
    config: AppConfig,
    narrative_sample: List[Dict[str, Any]],
    run_span=None,
) -> Dict[str, Any]:
    """
    Generate a formalized Reading Plan from a sample of historical
    SAR narratives with known FinCEN tags.
    """
    sample_text = ""
    for i, row in enumerate(narrative_sample[:10]):
        sample_text += f"\n--- Narrative {i+1} ---\n"
        sample_text += f"FinCEN Tags: {row.get('fincen_tags', 'N/A')}\n"
        sample_text += f"Narrative:\n{str(row.get('NARRATIVE', ''))[:3000]}\n"

    user_prompt = f"""
Here are {len(narrative_sample[:10])} historical SAR narratives with their 
FinCEN typology tags. Generate a Reading Plan that captures how to consistently 
extract and interpret information from SAR narratives for typology classification.

{sample_text}

Return JSON only with this structure:
{{
    "version": "v1",
    "sections": [
        {{
            "section_name": "string",
            "extraction_instructions": "string",
            "key_fields": ["string"]
        }}
    ],
    "behavioral_indicators": [
        {{
            "typology": "string",
            "indicators": ["string"]
        }}
    ],
    "contextual_signals": ["string"]
}}
"""

    t0 = time.time()
    response_text = await _call_gemini(
        instruction=READING_PLAN_PROMPT,
        user_prompt=user_prompt,
        model=config.model,
    )

    if run_span:
        run_span.log(metadata={
            "STEP": "generate_reading_plan",
            "latency_ms": _ms(t0)
        })

    # Clean response and parse JSON
    clean = response_text.strip()
    if clean.startswith("```"):
        clean = clean.split("```")[1]
        if clean.startswith("json"):
            clean = clean[4:]
    clean = clean.strip()

    return json.loads(clean)

# --- Rule Synthesis (Call 2) ---

SYNTHESIZE_RULES_PROMPT = """
You are an AML expert tasked with inductively synthesizing typology rules from
extracted SAR narrative structures. You will be given a new extracted narrative
structure along with the current candidate rule set. Your task is to either
reinforce existing rules where evidence converges, or introduce new behavioral
dimensions that broaden the typology's coverage.

Output a consolidated, non-overlapping set of rules expressed as conditional logic:
observable transaction features, quantified thresholds, and temporal constraints
that collectively characterize each typology's behavioral signature.

Return JSON only, no other text.
"""


async def synthesize_rules(
    config: AppConfig,
    extracted_structure: Dict[str, Any],
    fincen_tags: Dict[str, Any],
    existing_rules: Optional[List[Dict[str, Any]]],
    reading_plan: Dict[str, Any],
    run_span=None,
) -> Dict[str, Any]:
    """
    Call 2 of Rule Hypothesis Generation.
    Iterative inductive accumulation — takes one extracted narrative structure
    + current candidate rules, returns updated consolidated rules hypothesis.
    """
    existing_rules_text = json.dumps(existing_rules, indent=2) \
        if existing_rules else "No existing rules yet."

    user_prompt = f"""
Current candidate rules:
{existing_rules_text}

New extracted narrative structure:
{json.dumps(extracted_structure, indent=2)}

Ground truth FinCEN tags for this narrative:
{json.dumps(fincen_tags, indent=2)}

Reading Plan version: {reading_plan.get('version', 'v1')}

Update the rules hypothesis by reinforcing existing rules where this narrative
provides supporting evidence, or introducing new rules where new behavioral
patterns are observed. Return the complete consolidated rules hypothesis.

Return JSON only with this structure:
{{
    "version": "string",
    "typology_rules": [
        {{
            "typology": "string",
            "fincen_tag": "string",
            "confidence": 0.0,
            "rules": [
                {{
                    "rule_id": "string",
                    "condition": "string",
                    "threshold": "string",
                    "temporal_constraint": "string",
                    "supporting_case_count": 0
                }}
            ]
        }}
    ],
    "provenance": {{
        "narratives_processed": 0,
        "reading_plan_version": "string"
    }}
}}
"""

    t0 = time.time()
    response_text = await _call_gemini(
        instruction=SYNTHESIZE_RULES_PROMPT,
        user_prompt=user_prompt,
        model=config.model,
    )

    if run_span:
        run_span.log(metadata={
            "STEP": "synthesize_rules",
            "latency_ms": _ms(t0)
        })

    clean = response_text.strip()
    if clean.startswith("```"):
        clean = clean.split("```")[1]
        if clean.startswith("json"):
            clean = clean[4:]
    clean = clean.strip()

    return json.loads(clean)

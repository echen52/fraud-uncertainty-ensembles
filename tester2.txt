from dotenv import load_dotenv
load_dotenv('/home/jupyter/custom_agent/typology_agent/.env')

from tca.app_config import load_config
from tca.agent import TypologyClassificationAgent

config = load_config('/home/jupyter/custom_agent/typology_agent/config.yaml')
agent = TypologyClassificationAgent(config=config, mode='induct')

result = await agent.induct(
    start_date="2026-01-01",
    end_date="2026-03-31",
    limit=3
)

print(f"Status: {result['status']}")
print(f"Narratives processed: {result['narratives_processed']}")
print(f"Reading plan version: {result['reading_plan'].get('version')}")
print(f"Typologies found: {len(result['rules_hypothesis'].get('typology_rules', []))}")

async def synthesize_rules(
    config: AppConfig,
    extracted_structure: Dict[str, Any],
    fincen_tags: List[Dict[str, Any]],
    existing_rules: Optional[List[Dict[str, Any]]],
    pre_parse: Dict[str, Any],
    frames: Dict[str, Any],
    run_span=None,
) -> Dict[str, Any]:
    """
    Call 2 of Rule Hypothesis Generation.
    Iterative inductive accumulation — takes one extracted narrative
    structure + current candidate rules, returns updated consolidated
    rules hypothesis. Grounded by spaCy frames and pre_parse anchors.
    """
    existing_rules_text = json.dumps(existing_rules, indent=2) \
        if existing_rules else "No existing rules yet."

    user_prompt = f"""
Pre-extracted narrative signals (ground truth anchors):
{format_frames_for_prompt(frames)}

Suspicious total: {pre_parse.get('suspicious_total', 'Not found')}
Review period: {pre_parse.get('review_period', {}).get('start', 'N/A')} to {pre_parse.get('review_period', {}).get('end', 'N/A')}

Current candidate rules:
{existing_rules_text}

New extracted narrative structure:
{json.dumps(extracted_structure, indent=2)}

Active FinCEN tags for this narrative (one rule entry required per tag):
{json.dumps(fincen_tags, indent=2)}

Update the rules hypothesis. Use the pre-extracted signals above to ground
condition, threshold, temporal_constraint, and indicators fields.
Return the complete consolidated rules hypothesis.
"""

    t0 = time.time()
    response_text = await _call_gemini(
        instruction=SYNTHESIZE_RULES_PROMPT,
        user_prompt=user_prompt,
        model=config.model,
    )

    if run_span:
        run_span.log(metadata={
            "STEP": "synthesize_rules",
            "latency_ms": _ms(t0)
        })

    clean = response_text.strip()
    if clean.startswith("```"):
        clean = clean.split("```")[1]
        if clean.startswith("json"):
            clean = clean[4:]
    clean = clean.strip()

    return json.loads(clean)

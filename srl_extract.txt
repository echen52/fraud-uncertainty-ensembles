# tca/srl_extract.py
# Deterministic behavioral event extraction from SAR narrative text.
# Uses spaCy dependency parsing on Introduction and Conclusion sentences only.
# Runs before any LLM call. No network required after model load.

import re
import spacy
from typing import Optional

# Load once at module level — avoids reloading on every call
nlp = spacy.load("en_core_web_sm")

# Verbs that signal AML-relevant behavioral events
AML_ACTION_VERBS = {
    "deposit", "deposited", "deposits",
    "withdraw", "withdrew", "withdrawn", "withdrawals",
    "transfer", "transferred", "transfers",
    "send", "sent",
    "receive", "received",
    "wire", "wired",
    "fund", "funded",
    "disburse", "disbursed",
    "debit", "debited",
    "credit", "credited",
    "cash", "cashed",
    "purchase", "purchased",
    "move", "moved",
    "initiate", "initiated",
}

# Instrument keywords — maps to transaction channel
INSTRUMENT_KEYWORDS = {
    "zelle": "Zelle",
    "wire": "Wire Transfer",
    "ach": "ACH",
    "check": "Check",
    "cash": "Cash",
    "venmo": "Venmo",
    "cashapp": "Cash App",
    "cash app": "Cash App",
    "atm": "ATM",
    "square": "Square",
    "paypal": "PayPal",
    "official bank check": "Official Bank Check",
    "obc": "Official Bank Check",
}


def _split_intro_conclusion(narrative: str) -> tuple[str, str]:
    """Extract Introduction and Conclusion text from narrative."""
    intro_match = re.search(
        r'INTRODUCTION[:\s]*(.*?)(?=DETAILS OF INVESTIGATION|CONCLUSION|$)',
        narrative, re.IGNORECASE | re.DOTALL
    )
    conclusion_match = re.search(
        r'CONCLUSION[:\s]*(.*?)(?=CONTENTS OF FILE|$)',
        narrative, re.IGNORECASE | re.DOTALL
    )
    intro = intro_match.group(1).strip() if intro_match else ""
    conclusion = conclusion_match.group(1).strip() if conclusion_match else ""
    return intro, conclusion


def _extract_amount(token) -> Optional[str]:
    """
    Extract dollar amount from token subtree.
    Looks for $ sign + numeric siblings.
    """
    subtree_text = " ".join(t.text for t in token.subtree)
    match = re.search(r'\$[\d,]+\.?\d*', subtree_text)
    if match:
        return match.group(0)
    # Also check for numeric token with currency context
    for child in token.children:
        if child.like_num or child.is_currency:
            amount_match = re.search(
                r'\$?[\d,]+\.?\d*', child.text
            )
            if amount_match:
                return f"${amount_match.group(0).replace('$','')}"
    return None


def _extract_instrument(token) -> Optional[str]:
    """
    Extract transaction instrument/channel from token subtree.
    Checks subtree text against INSTRUMENT_KEYWORDS.
    """
    subtree_text = " ".join(
        t.text.lower() for t in token.subtree
    )
    # Check multi-word first
    for keyword, canonical in sorted(
        INSTRUMENT_KEYWORDS.items(), 
        key=lambda x: len(x[0]), 
        reverse=True
    ):
        if keyword in subtree_text:
            return canonical
    return None


def _extract_counterparty(verb_token) -> Optional[str]:
    """
    Extract counterparty from 'from' or 'to' prepositional phrases
    hanging off the verb.
    """
    for child in verb_token.children:
        if child.dep_ == "prep" and child.text.lower() in ("from", "to"):
            for pobj in child.children:
                if pobj.dep_ == "pobj":
                    # Get full noun phrase
                    phrase = " ".join(
                        t.text for t in pobj.subtree
                        if t.dep_ not in ("punct",)
                    )
                    return phrase.strip()
    return None


def _extract_subject(verb_token) -> Optional[str]:
    """Extract subject (actor) of verb."""
    for child in verb_token.children:
        if child.dep_ in ("nsubj", "nsubjpass"):
            return " ".join(
                t.text for t in child.subtree
                if t.dep_ not in ("punct",)
            ).strip()
    return None


def _extract_temporal(verb_token) -> Optional[str]:
    """Extract temporal modifier from verb subtree."""
    for child in verb_token.children:
        if child.dep_ in ("prep", "advmod", "npadvmod"):
            subtree_text = " ".join(
                t.text for t in child.subtree
            )
            # Check for time indicators
            if any(indicator in subtree_text.lower() for indicator in [
                "day", "month", "year", "week",
                "within", "between", "during", "period",
                "subsequent", "immediately", "rapidly"
            ]):
                return subtree_text.strip()
    return None


def extract_frames_from_sentence(sentence: str) -> list[dict]:
    """
    Run spaCy dependency parse on a single sentence.
    Returns list of AML behavioral event frames.
    """
    doc = nlp(sentence)
    frames = []

    for token in doc:
        # Only process AML-relevant verbs
        if token.lemma_.lower() not in AML_ACTION_VERBS:
            continue
        if token.pos_ not in ("VERB", "AUX"):
            continue

        frame = {
            "predicate": token.lemma_.lower(),
            "actor": _extract_subject(token),
            "amount": _extract_amount(token),
            "instrument": _extract_instrument(token),
            "counterparty": _extract_counterparty(token),
            "temporal": _extract_temporal(token),
            "sentence": sentence.strip()
        }

        # Only keep frames with at least predicate + one other field
        populated = sum(
            1 for k, v in frame.items()
            if k not in ("predicate", "sentence") and v is not None
        )
        if populated >= 1:
            frames.append(frame)

    return frames


def srl_extract(narrative: str) -> list[dict]:
    """
    Main entry point. Extracts behavioral event frames from
    Introduction and Conclusion sections of a SAR narrative.

    Returns list of frame dicts:
    {
        predicate: str,       # action verb (deposited, withdrawn, etc.)
        actor: str | None,    # who performed the action
        amount: str | None,   # dollar amount
        instrument: str | None, # transaction channel
        counterparty: str | None, # from/to entity
        temporal: str | None, # time constraint
        sentence: str         # source sentence for traceability
    }
    """
    intro, conclusion = _split_intro_conclusion(narrative)

    # Combine intro + conclusion, split into sentences
    combined = f"{intro} {conclusion}".strip()
    doc = nlp(combined)

    all_frames = []
    for sent in doc.sents:
        frames = extract_frames_from_sentence(sent.text)
        all_frames.extend(frames)

    # Deduplicate frames with identical predicate + amount
    seen = set()
    deduped = []
    for frame in all_frames:
        key = (frame["predicate"], frame["amount"])
        if key not in seen:
            seen.add(key)
            deduped.append(frame)

    return deduped


def format_frames_for_prompt(frames: list[dict]) -> str:
    """
    Formats extracted frames as structured text for injection
    into LLM prompt context.
    """
    if not frames:
        return "No behavioral events extracted."

    lines = ["Pre-extracted behavioral events:"]
    for i, frame in enumerate(frames, 1):
        lines.append(f"\nEvent {i}:")
        lines.append(f"  Action:       {frame['predicate']}")
        if frame['actor']:
            lines.append(f"  Actor:        {frame['actor']}")
        if frame['amount']:
            lines.append(f"  Amount:       {frame['amount']}")
        if frame['instrument']:
            lines.append(f"  Channel:      {frame['instrument']}")
        if frame['counterparty']:
            lines.append(f"  Counterparty: {frame['counterparty']}")
        if frame['temporal']:
            lines.append(f"  Temporal:     {frame['temporal']}")

    return "\n".join(lines)

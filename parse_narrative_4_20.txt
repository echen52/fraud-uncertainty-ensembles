# parse_narrative.py
# Pre-LLM deterministic extraction from SAR narrative text.
# Runs before any Gemini call to anchor quantifiable fields.

import re
from typing import Optional
from tca.typology_keywords import (
    SUSPICIOUS_TOTAL_PATTERNS,
    REVIEW_PERIOD_PATTERNS,
)


def _split_sections(narrative: str) -> dict:
    """Split narrative into Introduction and Conclusion sections."""
    sections = {}

    intro_match = re.search(
        r'INTRODUCTION[:\s]*(.*?)(?=DETAILS OF INVESTIGATION|CONCLUSION|$)',
        narrative, re.IGNORECASE | re.DOTALL
    )
    conclusion_match = re.search(
        r'CONCLUSION[:\s]*(.*?)(?=CONTENTS OF FILE|$)',
        narrative, re.IGNORECASE | re.DOTALL
    )

    sections["introduction"] = (
        intro_match.group(1).strip() if intro_match else ""
    )
    sections["conclusion"] = (
        conclusion_match.group(1).strip() if conclusion_match else ""
    )
    return sections


def extract_suspicious_total(narrative: str) -> Optional[float]:
    """
    Extract suspicious activity total.
    Priority: Conclusion first, then Introduction.
    """
    sections = _split_sections(narrative)
    search_order = [sections["conclusion"], sections["introduction"], narrative]

    for text in search_order:
        if not text:
            continue
        for pattern in SUSPICIOUS_TOTAL_PATTERNS:
            match = re.search(pattern, text, re.IGNORECASE)
            if match:
                amount_str = match.group(1).replace(",", "")
                try:
                    return float(amount_str)
                except ValueError:
                    continue
    return None


def extract_review_period(narrative: str) -> dict:
    """
    Extract review period start and end dates.
    Priority: Introduction first, then Conclusion.
    """
    sections = _split_sections(narrative)
    search_order = [sections["introduction"], sections["conclusion"]]

    for text in search_order:
        if not text:
            continue
        for pattern in REVIEW_PERIOD_PATTERNS:
            match = re.search(pattern, text, re.IGNORECASE)
            if match:
                return {
                    "start": match.group(1),
                    "end": match.group(2)
                }
    return {"start": None, "end": None}


def parse_narrative(narrative: str) -> dict:
    """
    Main entry point. Runs deterministic extraction of
    quantitative anchors before any LLM call.
    """
    return {
        "suspicious_total": extract_suspicious_total(narrative),
        "review_period": extract_review_period(narrative),
    }

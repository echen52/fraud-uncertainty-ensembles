# Excluded categories from v2 — irrelevant to retail banking
EXCLUDED_PREFIXES = [
    "FFI_F_EFC_",  # Embassy/Consulate
    "FFI_F_EMP_",  # Employee
    "FFI_F_BLK_",  # Bulk currency
    "FFI_F_CBF_",  # Cross-border casas de cambio
    "FFI_F_INS_",  # Insurance
    "FFI_F_LA_",   # Lending
    "FFI_F_TRD_",  # Trade Finance
]

# Filter vocabulary
filtered_vocab = {
    k: v for k, v in vocabulary.items()
    if not any(k.startswith(prefix) for prefix in EXCLUDED_PREFIXES)
}
print(f"Full vocabulary: {len(vocabulary)}")
print(f"Filtered vocabulary: {len(filtered_vocab)}")

# Embed filtered vocabulary once — reuse across all narratives
print("Embedding filtered vocabulary...")
filtered_ids = list(filtered_vocab.keys())
filtered_texts = list(filtered_vocab.values())
filtered_embeddings = batch_embed(filtered_texts)

THRESHOLD = 0.78

def match_signals_to_vocabulary(signals, label=""):
    print(f"\n{'='*70}")
    print(f"NARRATIVE: {label}")
    print(f"{'='*70}")
    
    sig_embeddings = batch_embed(signals)
    scores = cosine_similarity(sig_embeddings, filtered_embeddings)
    
    all_matches = []
    for i, signal in enumerate(signals):
        signal_scores = [
            (filtered_ids[j], scores[i][j]) 
            for j in range(len(filtered_ids)) 
            if scores[i][j] >= THRESHOLD
        ]
        signal_scores.sort(key=lambda x: x[1], reverse=True)
        
        print(f"\n  SIGNAL: \"{signal}\"")
        if signal_scores:
            for ind_id, score in signal_scores:
                print(f"    [{score:.3f}] {ind_id}: {filtered_vocab[ind_id][:65]}...")
            all_matches.extend(signal_scores)
        else:
            print(f"    No matches above {THRESHOLD}")
    
    # Summary
    unique_indicators = set(ind_id for ind_id, _ in all_matches)
    print(f"\n  SUMMARY: {len(all_matches)} total matches, "
          f"{len(unique_indicators)} unique indicators matched")
    return all_matches

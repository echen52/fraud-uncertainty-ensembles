# --- Top-K matcher ---
def match_signals_to_vocabulary(signals, label="", top_k=TOP_K, soft_floor=SOFT_FLOOR):
    print(f"\n{'='*70}")
    print(f"NARRATIVE: {label}")
    print(f"{'='*70}")

    sig_embeddings = batch_embed(signals)
    scores = cosine_similarity(sig_embeddings, filtered_embeddings)

    all_matches = []
    for i, signal in enumerate(signals):
        signal_scores = [
            (filtered_ids[j], float(scores[i][j]))
            for j in range(len(filtered_ids))
        ]
        # Sort descending, apply soft floor, take top K
        signal_scores.sort(key=lambda x: x[1], reverse=True)
        top_matches = [
            (ind_id, score) for ind_id, score in signal_scores[:top_k]
            if score >= soft_floor
        ]

        print(f"\n  SIGNAL: \"{signal}\"")
        if top_matches:
            for ind_id, score in top_matches:
                print(f"    [{score:.3f}] {ind_id}: {filtered_vocab[ind_id][:65]}...")
            all_matches.extend([
                {
                    "behavioral_signal": signal,
                    "indicator_id": ind_id,
                    "similarity_score": score
                }
                for ind_id, score in top_matches
            ])
        else:
            print(f"    No matches above soft floor ({soft_floor})")

    unique_indicators = set(m["indicator_id"] for m in all_matches)
    print(f"\n  SUMMARY: {len(all_matches)} total matches, "
          f"{len(unique_indicators)} unique indicators matched")
    return all_matches

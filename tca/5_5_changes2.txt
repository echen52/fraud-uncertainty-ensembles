WITH combo_cases AS (
SELECT DISTINCT ITEM_ID
FROM (
SELECT A.ITEM_ID, C.SUS_ACT_OTH as other_text
FROM (
SELECT s.CASE_ID AS ITEM_ID
FROM `esa-src-prod-5736.ACT_SRC_DATA.ACT_CLOUD_IMPL_V_SAR_FORM_STATUSES` s
INNER JOIN (
SELECT case_id, MAX(SNAPSHOT_DT) AS SNAPSHOT_DT
FROM `esa-src-prod-5736.ACT_SRC_DATA.ACT_CLOUD_IMPL_V_SAR_FORM_STATUSES`
WHERE AMENDMENT_OF IS NULL
GROUP BY case_id
) ss ON s.CASE_ID = ss.case_id AND s.SNAPSHOT_DT = ss.SNAPSHOT_DT
WHERE DATE(s.DATE_FILED_ENH) BETWEEN DATE ‘2025-01-01’ AND DATE ‘2025-12-31’
) A
LEFT JOIN `esa-prod-7486.ESA_MART_AML_VIEWS.ACT_SAR_BV` C ON A.ITEM_ID = C.CASE_ID
)
WHERE REGEXP_CONTAINS(other_text, r’(?i)rap[a-z]*\s*m[oa]?[a-z]*\s*(of\s*)?fund’)
AND REGEXP_CONTAINS(other_text, r’(?i)neg[a-z]*\s*news’)
)
SELECT COUNT(*) as combo_case_count FROM combo_cases

from tca.srl_extract import srl_extract, format_frames_for_prompt
from tca.tools.sar_narrative_tools import get_sar_narratives

CASE_IDS = [
    "AML_INV_00000051420",  # Joan Gaylord — cash deposit
    "AML_INV_00000055001",  # Brent Hemphill — OBC funnel
    "AML_INV_00000049396",  # Ylionet Doreus — Zelle P2P
]

df = get_sar_narratives(
    start_date="2026-01-01",
    end_date="2026-03-31"
)

for case_id in CASE_IDS:
    row = df[df["ITEM_ID"] == case_id]
    if row.empty:
        print(f"{case_id} — not found")
        continue

    narrative = row.iloc[0].get("NARRATIVE", "")
    frames = srl_extract(narrative)

    print(f"\n{'='*60}")
    print(f"Case: {case_id}")
    print(format_frames_for_prompt(frames))

import spacy
nlp = spacy.load("en_core_web_sm")

# Use the full Introduction paragraph
text = """This investigation was initiated as the result of an alert referred due to significant cash deposits with no known source of funds involving account 321130060684 in the names of Joan Gaylord and Jeffrey Gaylord. There were no prior investigations identified. This investigation, which covers the time period 07/01/2025 through 01/06/2026, disclosed suspicious account activity involving significant cash activity with no known source of funds. The suspicious activity totals $40,000.00 and occurred between 10/09/2025 and 10/14/2025."""

doc = nlp(text)

print("=== NAMED ENTITIES ===")
for ent in doc.ents:
    print(f"  {ent.text:30} label={ent.label_}")

print("\n=== NOUN CHUNKS ===")
for chunk in doc.noun_chunks:
    print(f"  {chunk.text:40} root={chunk.root.text}")

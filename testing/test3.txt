from __future__ import annotations

import argparse
import sys
from pathlib import Path

from tca.app_config import load_config
from tca.agent import TypologyClassificationAgent


def parse_args():
    parser = argparse.ArgumentParser(description="Typology Classification Agent")
    parser.add_argument(
        "--mode",
        choices=["induct", "classify", "eval"],
        required=True,
        help="Run mode: induct | classify | eval"
    )
    parser.add_argument(
        "--config",
        default="config.yaml",
        help="Path to config.yaml (default: config.yaml)"
    )
    parser.add_argument(
        "--start-date",
        default=None,
        help="Start date for data pull (YYYY-MM-DD). Overrides config default."
    )
    parser.add_argument(
        "--end-date",
        default=None,
        help="End date for data pull (YYYY-MM-DD). Overrides config default."
    )
    parser.add_argument(
        "--narrative",
        default=None,
        help="SAR narrative text for classify mode"
    )
    parser.add_argument(
        "--case-id",
        default=None,
        help="Case ID for classify/eval mode"
    )
    parser.add_argument(
        "--fincen-tag",
        default=None,
        help="Ground truth FinCEN tag for eval mode"
    )
    return parser.parse_args()


def run_induct(agent: TypologyClassificationAgent, args):
    print(f"Starting induction pipeline...")
    print(f"Date range: {args.start_date or agent.config.default_start_date} "
          f"to {args.end_date or agent.config.default_end_date}")

    result = agent.induct(
        start_date=args.start_date,
        end_date=args.end_date,
    )

    if result["status"] == "error":
        print(f"Induction failed: {result['message']}")
        sys.exit(1)

    print(f"\nInduction complete.")
    print(f"Narratives processed: {result['narratives_processed']}")
    typology_rules = result["rules_hypothesis"].get("typology_rules", [])
    print(f"Typologies in Rules Hypothesis: {len(typology_rules)}")
    for t in typology_rules:
        print(f"  - {t.get('typology')} ({len(t.get('rules', []))} rules, "
              f"confidence: {t.get('confidence', 0):.2f})")


def run_classify(agent: TypologyClassificationAgent, args):
    if not args.narrative and not args.case_id:
        print("Error: classify mode requires --narrative or --case-id")
        sys.exit(1)

    narrative = args.narrative

    # If case_id provided instead of raw narrative, 
    # pull it from BQ
    if not narrative and args.case_id:
        from tca.tools.sar_narrative_tools import get_sar_narratives
        print(f"Fetching narrative for case {args.case_id}...")
        df = get_sar_narratives(case_id=args.case_id)
        if len(df) == 0:
            print(f"No narrative found for case_id: {args.case_id}")
            sys.exit(1)
        narrative = df.iloc[0]["NARRATIVE"]

    print(f"Classifying case: {args.case_id or 'ad-hoc narrative'}")

    result = agent.classify(
        narrative=narrative,
        case_id=args.case_id,
        fincen_tag_ground_truth=args.fincen_tag,
    )

    if result["status"] == "error":
        print(f"Classification failed: {result['message']}")
        sys.exit(1)

    print(f"\nClassification complete for case: {result['case_id']}")
    print(f"Ranked typology candidates:")
    for i, candidate in enumerate(result["ranked_candidates"], 1):
        print(f"  {i}. {candidate.get('typology')} "
              f"(confidence: {candidate.get('confidence', 0):.2f})")
        print(f"     FinCEN tag: {candidate.get('fincen_tag', 'N/A')}")
        print(f"     Reasoning: {candidate.get('reasoning', 'N/A')[:200]}")


def run_eval(agent: TypologyClassificationAgent, args):
    """
    Eval mode — runs classify against a held-out set pulled from BQ,
    logs expected vs actual to Braintrust for scoring.
    """
    from tca.tools.sar_narrative_tools import get_sar_narratives
    from tca.tools.fincen_tag_tools import get_fincen_tags

    print(f"Starting eval pipeline...")
    print(f"Date range: {args.start_date or agent.config.default_start_date} "
          f"to {args.end_date or agent.config.default_end_date}")

    # Pull held-out narratives + ground truth tags
    df_narratives = get_sar_narratives(
        start_date=args.start_date or agent.config.default_start_date,
        end_date=args.end_date or agent.config.default_end_date,
    )
    df_tags = get_fincen_tags(
        start_date=args.start_date or agent.config.default_start_date,
        end_date=args.end_date or agent.config.default_end_date,
    )
    df_joined = df_narratives.merge(df_tags, on="ITEM_ID", how="inner")

    print(f"Evaluating {len(df_joined)} held-out cases...")

    results = []
    for _, row in df_joined.iterrows():
        case_id = row["ITEM_ID"]
        narrative = row.get("NARRATIVE", "")

        if not narrative or len(str(narrative)) < 100:
            continue

        # Build ground truth label from FinCEN tags
        ground_truth_tags = {
            "STRCTNG": row.get("STRCTNG"),
            "MONEY_LAUNDERING": row.get("MONEY_LAUNDERING"),
            "FRAUD": row.get("FRAUD"),
            "TERROR_FINANCING": row.get("TERROR_FINANCING"),
            "CYBER": row.get("CYBER"),
        }
        # Use first non-null tag as primary ground truth label
        primary_tag = next(
            (k for k, v in ground_truth_tags.items() if v and str(v).strip()),
            None
        )

        result = agent.classify(
            narrative=str(narrative),
            case_id=case_id,
            fincen_tag_ground_truth=primary_tag,
        )
        results.append({
            "case_id": case_id,
            "ground_truth": primary_tag,
            "top_prediction": result["ranked_candidates"][0].get("fincen_tag")
            if result["ranked_candidates"] else None,
            "ranked_candidates": result["ranked_candidates"],
        })

    # Summary stats
    correct = sum(
        1 for r in results
        if r["top_prediction"] == r["ground_truth"]
    )
    total = len(results)
    accuracy = correct / total if total > 0 else 0

    print(f"\nEval complete.")
    print(f"Cases evaluated: {total}")
    print(f"Top-1 accuracy: {accuracy:.2%}")
    print(f"Results logged to Braintrust experiment: {agent.branch_exp}")


def main():
    args = parse_args()

    # Load config
    config = load_config(args.config)

    # Instantiate agent
    agent = TypologyClassificationAgent(config=config, mode=args.mode)

    if args.mode == "induct":
        run_induct(agent, args)
    elif args.mode == "classify":
        # Must induct first to load rules hypothesis
        print("Running induction before classification...")
        agent.induct(
            start_date=args.start_date,
            end_date=args.end_date,
        )
        run_classify(agent, args)
    elif args.mode == "eval":
        # Must induct first to load rules hypothesis
        print("Running induction before eval...")
        agent.induct(
            start_date=args.start_date,
            end_date=args.end_date,
        )
        run_eval(agent, args)


if __name__ == "__main__":
    main()

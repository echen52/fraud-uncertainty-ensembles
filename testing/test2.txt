from __future__ import annotations

import os
import subprocess
import time
import uuid
from datetime import datetime
from zoneinfo import ZoneInfo
from typing import Any, Dict, List, Optional

import braintrust
from braintrust.wrappers.google_genai import setup_genai
from google import genai
from google.cloud import bigquery

from .app_config import AppConfig
from .tools.sar_narrative_tools import get_sar_narratives
from .tools.fincen_tag_tools import get_fincen_tags
from .llm_helpers import (
    build_gemini_config,
    generate_reading_plan,
    extract_narrative_structure,
    synthesize_rules,
    classify_narrative,
)


def _ms(t0: float) -> int:
    return int((time.time() - t0) * 1000)


def _get_branch() -> str:
    branch = subprocess.run(
        ['git', 'rev-parse', '--abbrev-ref', 'HEAD'],
        capture_output=True, text=True, check=True
    )
    return branch.stdout.strip()


class TypologyClassificationAgent:
    def __init__(self, config: AppConfig, mode: str):
        """
        mode: 'induct' | 'classify' | 'eval'
        """
        self.config = config
        self.mode = mode
        self.bq = bigquery.Client(project=self.config.bq_project_sar)

        # Braintrust setup
        self.bt_project = config.bt_project
        self.bt_api_key = os.getenv("BRAINTRUST_API_KEY")
        setup_genai(project_name=self.bt_project, api_key=self.bt_api_key)

        self.branch = _get_branch()
        self.branch_exp = f"branch/{self.branch}"
        self.bt = braintrust.init(
            project=self.bt_project,
            experiment=self.branch_exp,
            update=True
        )

        self.run_id: Optional[str] = None
        self.run_span: Optional[Any] = None

        # Vertex AI routing via env vars
        use_vertexai = os.getenv("GOOGLE_GENAI_USE_VERTEXAI", "false").lower() == "true"
        project_id = os.getenv("GOOGLE_CLOUD_PROJECT")
        location = os.getenv("GOOGLE_CLOUD_LOCATION")

        if use_vertexai:
            self.client = genai.Client(
                vertexai=True, project=project_id, location=location
            )
        else:
            self.client = genai.Client()

        # Rules hypothesis and reading plan — populated during
        # induct, used during classify
        self.rules_hypothesis: List[Dict[str, Any]] = []
        self.reading_plan: Dict[str, Any] = {}

    # --- Run span helper ---
    def _start_run_span(
        self,
        label: str,
        input_data: Dict,
        expected: Optional[Dict] = None
    ):
        ts = datetime.now(tz=ZoneInfo("America/New_York")).strftime("%Y%m%d-%H%M")
        self.run_id = f"{label}_{ts}-{uuid.uuid4().hex[:8]}"
        self.run_span = self.bt.start_span(name=self.run_id)
        log_kwargs = dict(
            input=input_data,
            metadata={
                "BRANCH": self.branch,
                "RUN_ID": self.run_id,
                "MODE": self.mode
            },
            tags=[self.branch, self.run_id],
        )
        if expected:
            log_kwargs["expected"] = expected
        self.run_span.log(**log_kwargs)

    # --- Mode: induct ---
    def induct(
        self,
        start_date: str = None,
        end_date: str = None,
    ) -> Dict[str, Any]:
        """
        Pull historical SAR narratives with known FinCEN tags,
        build Reading Plan, inductively accumulate Rules Hypothesis.
        """
        start_date = start_date or self.config.default_start_date
        end_date = end_date or self.config.default_end_date

        self._start_run_span(
            label="induct",
            input_data={"start_date": start_date, "end_date": end_date}
        )

        # Step 1: Fetch SAR narratives
        with self.run_span.start_span(name="induct.fetch_narratives") as span:
            t0 = time.time()
            df_narratives = get_sar_narratives(
                start_date=start_date,
                end_date=end_date
            )
            span.log(
                output={"row_count": len(df_narratives)},
                metrics={"latency_ms": _ms(t0)},
                metadata={"STEP": "fetch_narratives"}
            )

        # Step 2: Fetch FinCEN tags
        with self.run_span.start_span(name="induct.fetch_fincen_tags") as span:
            t0 = time.time()
            df_tags = get_fincen_tags(
                start_date=start_date,
                end_date=end_date
            )
            span.log(
                output={"row_count": len(df_tags)},
                metrics={"latency_ms": _ms(t0)},
                metadata={"STEP": "fetch_fincen_tags"}
            )

        # Step 3: Join on ITEM_ID — this is the labeled training corpus
        df_joined = df_narratives.merge(df_tags, on="ITEM_ID", how="inner")
        self.run_span.log(
            metadata={"JOINED_ROW_COUNT": len(df_joined)}
        )

        if len(df_joined) == 0:
            self.run_span.log(
                metadata={"STATUS": "error", "MESSAGE": "No joined rows — check date range or ITEM_ID alignment"}
            )
            return {"status": "error", "message": "No joined rows"}

        # Step 4: Generate Reading Plan from narrative sample
        with self.run_span.start_span(name="induct.generate_reading_plan") as span:
            t0 = time.time()
            narrative_sample = df_joined[["ITEM_ID", "NARRATIVE"]].head(10).to_dict("records")
            # Attach fincen tags to sample for context
            for row in narrative_sample:
                item_id = row["ITEM_ID"]
                tag_row = df_tags[df_tags["ITEM_ID"] == item_id]
                row["fincen_tags"] = tag_row[["STRCTNG", "MONEY_LAUNDERING", "FRAUD",
                                              "TERROR_FINANCING", "CYBER"]].to_dict("records")[0] \
                                     if len(tag_row) > 0 else {}

            self.reading_plan = generate_reading_plan(
                client=self.client,
                config=self.config,
                narrative_sample=narrative_sample,
                run_span=self.run_span,
            )
            span.log(
                output={"reading_plan_version": self.reading_plan.get("version", "v1")},
                metrics={"latency_ms": _ms(t0)},
                metadata={"STEP": "generate_reading_plan"}
            )

        # Step 5: Iteratively extract + synthesize rules across all narratives
        narratives_processed = 0
        for _, row in df_joined.iterrows():
            case_id = row["ITEM_ID"]
            narrative = row.get("NARRATIVE", "")

            if not narrative or len(str(narrative)) < 100:
                continue

            # Build fincen tag dict for this case
            fincen_tags = {
                "STRCTNG": row.get("STRCTNG"),
                "MONEY_LAUNDERING": row.get("MONEY_LAUNDERING"),
                "FRAUD": row.get("FRAUD"),
                "TERROR_FINANCING": row.get("TERROR_FINANCING"),
                "CYBER": row.get("CYBER"),
                "OTH_SUS_ACTIVITY": row.get("OTH_SUS_ACTIVITY"),
                "STRCTNG_DESC": row.get("STRCTNG_DESC"),
                "MONEY_LAUNDERING_DESC": row.get("MONEY_LAUNDERING_DESC"),
                "FRAUD_DESC": row.get("FRAUD_DESC"),
            }

            # Call 1: Extract narrative structure
            with self.run_span.start_span(name=f"induct.extract.{case_id}") as span:
                t0 = time.time()
                extracted = extract_narrative_structure(
                    client=self.client,
                    config=self.config,
                    narrative=str(narrative),
                    case_id=case_id,
                    reading_plan=self.reading_plan,
                    run_span=self.run_span,
                )
                span.log(
                    metrics={"latency_ms": _ms(t0)},
                    metadata={"STEP": "extract_narrative_structure", "CASE_ID": case_id}
                )

            # Call 2: Synthesize rules inductively
            with self.run_span.start_span(name=f"induct.synthesize.{case_id}") as span:
                t0 = time.time()
                self.rules_hypothesis = synthesize_rules(
                    client=self.client,
                    config=self.config,
                    extracted_structure=extracted,
                    fincen_tags=fincen_tags,
                    existing_rules=self.rules_hypothesis,
                    reading_plan=self.reading_plan,
                    run_span=self.run_span,
                )
                span.log(
                    metrics={"latency_ms": _ms(t0)},
                    metadata={"STEP": "synthesize_rules", "CASE_ID": case_id}
                )

            narratives_processed += 1

        self.run_span.log(
            output={
                "narratives_processed": narratives_processed,
                "rules_hypothesis_typologies": len(
                    self.rules_hypothesis.get("typology_rules", [])
                )
            },
            metadata={"STATUS": "success"}
        )
        self.bt.flush()

        return {
            "status": "success",
            "narratives_processed": narratives_processed,
            "rules_hypothesis": self.rules_hypothesis,
            "reading_plan": self.reading_plan,
        }

    # --- Mode: classify ---
    def classify(
        self,
        narrative: str,
        case_id: str = None,
        fincen_tag_ground_truth: str = None,
    ) -> Dict[str, Any]:
        """
        Classify an unseen SAR narrative against the Rules Hypothesis.
        Returns ranked typology candidates with confidence scores.
        If fincen_tag_ground_truth provided (eval mode),
        logs expected to Braintrust.
        """
        if not self.rules_hypothesis:
            return {
                "status": "error",
                "message": "No rules hypothesis loaded. Run induct() first."
            }

        expected = {"fincen_tag": fincen_tag_ground_truth} \
            if fincen_tag_ground_truth else None

        self._start_run_span(
            label=f"classify_{case_id or 'unknown'}",
            input_data={"narrative": narrative[:500], "case_id": case_id},
            expected=expected,
        )

        with self.run_span.start_span(name="classify.score_typologies") as span:
            t0 = time.time()
            result = classify_narrative(
                client=self.client,
                config=self.config,
                narrative=narrative,
                case_id=case_id,
                rules_hypothesis=self.rules_hypothesis,
                reading_plan=self.reading_plan,
                run_span=self.run_span,
            )
            span.log(
                output={"ranked_candidates": result.get("ranked_candidates", [])},
                metrics={"latency_ms": _ms(t0)},
                metadata={"STEP": "classify_narrative", "CASE_ID": case_id}
            )

        self.run_span.log(
            output={"ranked_candidates": result.get("ranked_candidates", [])}
        )
        self.bt.flush()

        return {
            "status": "success",
            "case_id": case_id,
            "ranked_candidates": result.get("ranked_candidates", []),
        }
